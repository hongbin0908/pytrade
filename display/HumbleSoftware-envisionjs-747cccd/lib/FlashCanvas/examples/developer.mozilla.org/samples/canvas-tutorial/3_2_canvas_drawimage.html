Canvas drawImage example 2

function draw() {
  var ctx = document.getElementById('canvas').getContext('2d');
  var img = new Image();
  img.onload = function(){
    for (i=0;i<4;i++){
      for (j=0;j<3;j++){
        ctx.drawImage(img,j*50,i*38,50,38);
      }
    }

    // prevent IE6 memory leak
    img = null;
  }
  img.src = 'images/rhino.jpg';
}
